\defbeamertemplate{footline}{progress bar}{
  \ifnum\thepage>1\relax%
  \dimen0=\paperwidth
  \multiply\dimen0 by \insertframenumber
  \divide\dimen0 by \inserttotalframenumber
  \edef\progressbarwidth{\the\dimen0}

  \leavevmode%
  \begin{beamercolorbox}[wd=\paperwidth,ht=0.5ex,dp=0.5ex]{progress bar}
    \begin{beamercolorbox}[wd=\progressbarwidth,ht=0.5ex,dp=0.5ex]{progress bar progress}
    \end{beamercolorbox}%
  \end{beamercolorbox}%
  \fi
}
